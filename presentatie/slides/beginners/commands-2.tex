\begin{frame}[fragile]{LaTeX commnands}
    Some commands go in the \textbf{body} of the document
    \newline
    \begin{itemize}
    \item The command \mintinline{tex}{\LaTeX} prints the latex logo and goes in the \textbf{body} of the document. \newline

    \item \mintinline{tex}{\newpage} starts a new page and is also goes in the \textbf{body} of the document.
    \newline

    \item \mintinline{tex}{\textbf{text}} is a command for \textbf{bold} text. The command takes 1 argument.
    \newline
    \item \mintinline{tex}{\sqrt[3]{y}}  \(\sqrt[3]{y}\) the square root command takes 1 argument and 1 optional argument
    \end{itemize}
\end{frame}
