\begin{frame}[fragile]{LaTeX commands}
    Other commands go in the \textbf{preamble} of the document
    \newline
    \begin{itemize}
    \item \mintinline{tex}{\title} sets the title of the document.\newline


    \item \mintinline{tex}{\usepackage{PACKAGENAME}} loads LaTeX code from other authors into your document. This code will often define new commands or tweak existing commands
    \newline

    \item \mintinline{tex}{\usepackage[paper=a5paper, margin=2cm, landscape=true]{geometry} } 
        loads the geometry package with 3 optional arguments
    \end{itemize}
\end{frame}
