\def\uncoverbox#1#2#3#4{%
	\def\opt{padding=2pt,left=#2}%
	\unless\ifishandout
	\only<-\numexpr #1-1\relax>{\edef\opt{\opt,bgcolor=lightgray}}%
	\fi
	\adjustbox{left={#3}}{%
		\expandafter\adjustbox\expandafter{\opt}{%
			\uncover<#1->{#4}%
		}%
	}%
}
