\begin{frame}[fragile]{LaTeX commands}
    Andere commando's staan in de \textbf{preamble} van het document
    \newline
    \begin{itemize}
    \item Met \mintinline{tex}{\title} geef je het document een titel.\newline

    \item \mintinline{tex}{\usepackage{...}} laadt LaTeX code van anderen in je document. Deze code defini\"eert vaak nieuwe commando's of past bestaande commando's aan. Soms verandert de opmaak van je pagina ook door het laden van een package.
    \newline

    \item \mintinline{tex}{\usepackage[paper=a5paper, margin=2cm, landscape=true]{geometry} } 
        laadt het geometry package met 3 optionele argumenten.
    \end{itemize}
\end{frame}
