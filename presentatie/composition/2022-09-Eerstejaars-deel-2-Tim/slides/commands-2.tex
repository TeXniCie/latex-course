\begin{frame}[fragile]{LaTeX commands}
    Sommige commando's staan in de \textbf{body} van het document
    \newline
    \begin{itemize}
    \item Het commando \mintinline{tex}{\LaTeX} print het \LaTeX{} logo. Dit commando staat in de \textbf{body} van het document. \newline

    \item \mintinline{tex}{\newpage} begint een nieuwe pagina en staat ook in de \textbf{body} van het document.
    \newline

    \item \mintinline{tex}{\textbf{text}} is een commando voor \textbf{vetgedrukte} tekst. Dit commando heeft 1 argument.
    \newline
    \item \mintinline{tex}{\sqrt[3]{y}}  \(\sqrt[3]{y}\) het wortelargument heeft 1 argument en 1 optioneel argument.
    \end{itemize}
\end{frame}
